72  public final CodedPiece[] canSolve() {
73  Thread currThread = Thread.currentThread();
74  int initialPriority = currThread.getPriority();
75  Thread.currentThread().setPriority(Thread.MAX_PRIORITY);
76  CodedPiece[] solvingCodedSlices = null;
77  synchronized (_lock) {
78  if (_solvingCodedSlices != null)
79  return null;
80 
81  int requiredSlices = getRequiredCodedPieceCount();
82  int receivedSlices = _cbbs.size();
83 
84  for (int retry = 0; retry <= receivedSlices - requiredSlices; retry++) {
85  if (retry != 0)
86  System.out.println("RETRYING TO DECODE: " + retry + "[" + _cbbs.size() + "-" + requiredSlices + "]");
87 
88  solvingCodedSlices = canSolveHighPriority();
89  if (solvingCodedSlices != null)
90  break;
91  receivedSlices = _cbbs.size();
92  }
93  }
94  Thread.currentThread().setPriority(initialPriority);
95 
96  return solvingCodedSlices;
97  }

150  @Override
151  public boolean decode() {
152  synchronized (_lock) {
153  if (_solvingCodedSlices != null)
154  return true;
155  }
156  CodedPiece[] solvingCodedSlices = canSolve();
157  if (solvingCodedSlices == null)
158  return false;
159 
160  // Now solve the equations
161  int slicesCount = solvingCodedSlices.length;
162  Byte[][] coefficientsArray = new Byte[slicesCount][];
163  SliceMatrix[] slicesContent = new SliceMatrix[slicesCount];
164  for (int i = 0; i < slicesCount; i++) {
165  slicesContent[i] = solvingCodedSlices[i]._codedContent;
166  coefficientsArray[i] = solvingCodedSlices[i]._cc.getCoefficients();
167  }
168 
169  BulkMatrix contentMatrix = createNewBulkMatrix(slicesContent);
170  ByteMatrix coefficientsMatrix = new ByteMatrix(coefficientsArray);
171 
172  ByteMatrix coefficientsMatrixInverse = (ByteMatrix) coefficientsMatrix.inverseMatrix();
173  if (coefficientsMatrixInverse == null)
174  return false;
175 
176  coefficientsMatrixInverse = _inverse;
177  return decodeUsing(coefficientsMatrixInverse, contentMatrix);
178  }

234  protected List<Integer> getRandomIndex(int maxIndex, int maxReturnSize) {
235  if (maxReturnSize > maxIndex)
236  throw new IllegalArgumentException("Invalid arguments: " + maxReturnSize + " vs. " + maxIndex);
237 
238  List<Integer> ret = new LinkedList<Integer>();
239  if (maxReturnSize == 0)
240  return ret;
241 
242  for (int i = 0; i < maxIndex; i++)
243  ret.add(new Integer(i));
244 
245  for (int i = maxIndex; i > maxReturnSize; i--) {
246  int removedIndex = _rand.nextInt(i);
247  ret.remove(removedIndex);
248  if (ret.size() != i - 1)
249  throw new IllegalStateException();
250  }
251 
252  return ret;
253  }

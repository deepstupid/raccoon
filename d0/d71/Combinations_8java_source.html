63  protected void generateBVRecursively(Vector<BitSet> allCombinations, BitSet bv, int bitIndex, int bitsSet, int bitsRequired, int total) {
64  if (bitIndex > total)
65  return;
66 
67  if (total - bitIndex < bitsRequired)
68  return;
69 
70  if (bitsRequired == 0) {
71  allCombinations.add((BitSet) bv.clone());
72  return;
73  }
74 
75  bv.set(bitIndex);
76  generateBVRecursively(allCombinations, bv, bitIndex + 1, bitsSet + 1, bitsRequired - 1, total);
77 
78  bv.clear(bitIndex);
79  generateBVRecursively(allCombinations, bv, bitIndex + 1, bitsSet, bitsRequired, total);
80  }

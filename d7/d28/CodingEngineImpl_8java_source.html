154  @Override
155  protected void processCodingEvent(CodingEngineEvent event) {
156  super.processCodingEvent(event);
157 
158  switch(event._eventType){
159  case ENG_ET_NEW_TASK:
160  {
161  CodingEngineEvent_NewCodingTask codingEvent = (CodingEngineEvent_NewCodingTask) event;
162  CodingTask cTask = codingEvent._cTask;
163  scheduleTask(cTask);
164  break;
165  }
166 
167  case ENG_ET_TASK_FAILED:
168  {
169  CodingEngineEvent_ExecutionEvent execEvent = (CodingEngineEvent_ExecutionEvent) event;
170  CodingTask cTask = execEvent._cTask;
171  CodingTaskStatus status = cTask.getStatus();
172  assert status == CodingTaskStatus.FAILED;
173  ICodingListener listener = cTask._listener;
174  listener.codingFailed(cTask._result);
175  break;
176  }
177 
178  case ENG_ET_TASK_FINISHED:
179  {
180  CodingEngineEvent_ExecutionEvent execEvent = (CodingEngineEvent_ExecutionEvent) event;
181  CodingTask cTask = execEvent._cTask;
182  CodingTaskStatus status = cTask.getStatus();
183  assert status == CodingTaskStatus.FINISHED;
184  ICodingListener listener = cTask._listener;
185  listener.codingFinished(cTask._result);
186  break;
187  }
188 
189  case ENG_ET_TASK_STARTED:
190  {
191  CodingEngineEvent_ExecutionEvent execEvent = (CodingEngineEvent_ExecutionEvent) event;
192  CodingTask cTask = execEvent._cTask;
193  ICodingListener listener = cTask._listener;
194  listener.codingStarted(cTask._result);
195  break;
196  }
197 
198  case ENG_ET_THREAD_NEW:
199  {
200  CodingEngineEvent_NewThreadEvent tEvent = (CodingEngineEvent_NewThreadEvent) event;
201  CodingThread cThread = tEvent._cThread;
202  threadAdded(cThread);
203  break;
204  }
205 
206  case ENG_ET_THREAD_FREE:
207  {
208  CodingEngineEvent_FreeThreadEvent tEvent = (CodingEngineEvent_FreeThreadEvent) event;
209  CodingThread cThread = tEvent._cThread;
210  threadBecameFree(cThread);
211  break;
212  }
213 
214  case ENG_ET_THREAD_BUSY:
215  {
216  CodingEngineEvent_BusyThreadEvent tEvent = (CodingEngineEvent_BusyThreadEvent) event;
217  CodingThread cThread = tEvent._cThread;
218  threadBecameBusy(cThread);
219  break;
220  }
221 
222  case END_ET_SEQ_TASK_FAILED:
223  {
224  CodingEngineEvent_SequentialCodingTaskFailed seqEvent = (CodingEngineEvent_SequentialCodingTaskFailed) event;
225  SequentialCodingTask seqCodingTask = seqEvent._seqCodingTask;
226  notifyListener(seqCodingTask);
227  break;
228  }
229 
230  case END_ET_SEQ_TASK_FINISHED:
231  {
232  CodingEngineEvent_SequentialCodingTaskFinished seqEvent = (CodingEngineEvent_SequentialCodingTaskFinished) event;
233  SequentialCodingTask seqCodingTask = seqEvent._seqCodingTask;
234  notifyListener(seqCodingTask);
235  break;
236  }
237 
238  default:
239  throw new UnsupportedOperationException("Unknown event type: " + event);
240  }
241  }
